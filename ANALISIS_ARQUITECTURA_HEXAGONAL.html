2️⃣ La Arquitectura Hexagonal (Puertos y Adaptadores)

Concepto Fundamental

La Arquitectura Hexagonal (también conocida como Puertos y Adaptadores) es un patrón arquitectónico que aísla la lógica de negocio del mundo exterior a través de interfaces abstractas.

Los Seis Pilares 🏛️

1. Núcleo de Dominio

Lógica de negocio pura, sin dependencias externas. Objetos que representan conceptos reales de tu dominio (HardwareItem, Category, Inventory).

Regla: Cero dependencias de frameworks

2. Puertos (Interfaces)

Contratos que definen cómo el mundo exterior puede usar tu dominio. Especifican qué debe implementar cada adaptador (HardwareRepositoryPort, EmailServicePort).

Regla: Son abstracciones, no implementaciones

3. Adaptadores Primarios

Controlan la entrada al dominio. Pueden ser HTTP (FastAPI), CLI, gRPC. Traducen requests externos a llamadas de dominio.

Ejemplo: Controllers, Controllers, Handlers

4. Adaptadores Secundarios

El dominio los usa para comunicarse con sistemas externos. Implementan puertos que define el dominio (Repositories, External APIs, Message Queues).

Ejemplo: DB, Email, Payment Gateways

5. Inversión de Control

El dominio NO conoce sus puertos. Los puertos se inyectan en tiempo de ejecución. Permite cambiar implementaciones sin tocar la lógica.

Beneficio: Testeable, flexible, mantenible

6. Independencia de Frameworks

Puedes cambiar de FastAPI a Flask a gRPC sin tocar el dominio. Solo cambian los adaptadores primarios.

Ventaja: Future-proof, reutilizable

Capas en Orden de Importancia 📊

  1. Núcleo de Dominio (Sagrado)
    Es lo más importante. Aquí está tu oro. Debe ser independiente, puro, testeable. 0 dependencias externas.
  2. Puertos (Contratos)
    Define interfaces que todos deben cumplir. Si cambias la BD, el puerto sigue igual. El adaptador cambia.
  3. Adaptadores Secundarios (Implementaciones)
    SQL, APIs externas, email. Pueden cambiar. El dominio no se entera.
  4. Adaptadores Primarios (API)
    FastAPI, CLI, gRPC. Son entrypoints. Pueden cambiar también.

// domain/entities/hardware.py
from dataclasses import dataclass
from typing import Optional

@dataclass
class HardwareEntity:
    """Entidad de dominio pura. Cero dependencias a frameworks."""
    nombre: str
    categoria: str  # CategoriaValueObject en versión completa
    stock: int
    precio: float
    
    def __post_init__(self):
        # Invariantes del dominio
        if self.stock < 0:
            raise ValueError("Stock no puede ser negativo")
        if self.precio <= 0:
            raise ValueError("Precio debe ser positivo")
        if not self.nombre.strip():
            raise ValueError("Nombre req uerido")
    
    def reducir_stock(self, cantidad: int) -> 'HardwareEntity':
        """Valida y crea nueva entidad con stock reducido."""
        if cantidad > self.stock:
            raise InsufficientStockError(
                f"No hay suficiente stock. Disponible: {self.stock}, solicitado: {cantidad}"
            )
        return HardwareEntity(
            nombre=self.nombre,
            categoria=self.categoria,
            stock=self.stock - cantidad,
            precio=self.precio,
        )
    
    def es_bajo_stock(self, umbral: int = 10) -> bool:
        """Business rule: considera bajo stock < umbral."""
        return self.stock < umbral
    
    def valor_inventario(self) -> float:
        """Calcula valor total de este item."""
        return self.stock * self.precio

// domain/ports/repository_port.py
from abc import ABC, abstractmethod
from typing import List, Optional
from domain.entities.hardware import HardwareEntity

class HardwareRepositoryPort(ABC):
    """Contrato que cualquier repositorio debe cumplir."""
    
    @abstractmethod
    async def guardar(self, entity: HardwareEntity) -> int:
        """Guarda una entidad. Retorna el ID asignado."""
        pass
    
    @abstractmethod
    async def obtener(self, id: int) -> Optional[HardwareEntity]:
        """Obtiene una entidad por ID."""
        pass
    
    @abstractmethod
    async def obtener_todos(self, skip: int = 0, limit: int = 100
                          ) -> List[HardwareEntity]:
        """Obtiene múltiples entidades con paginación."""
        pass
    
    @abstractmethod
    async def obtener_por_categoria(self, categoria: str) -> List[HardwareEntity]:
        """Obtiene items de una categoría."""
        pass
    
    @abstractmethod
    async def buscar_por_nombre(self, nombre: str) -> List[HardwareEntity]:
        """Busca items cuyo nombre contenga el texto."""
        pass

Paso 4: Servicio de Dominio (Casos de Uso)

// domain/services/hardware_service.py
class HardwareService:
    """Lógica de negocio. Implementación de casos de uso."""
    
    def __init__(self, repository: HardwareRepositoryPort):
        self.repository = repository  # ← Inyección de puerto
    
    async def crear_item(self, nombre: str, categoria: str, 
                        stock: int, precio: float) -> HardwareEntity:
        """Caso de uso: crear un nuevo item de hardware."""
        
        # Validación de dominio
        entity = HardwareEntity(nombre, categoria, stock, precio)
        
        # Persiste usando puerto
        item_id = await self.repository.guardar(entity)
        entity.id = item_id
        
        # En versión completa, emitirías un evento:
        # self._emit_event(HardwareCreatedEvent(entity))
        
        return entity
    
    async def reducir_stock_item(self, item_id: int, cantidad: int) -> HardwareEntity:
        """Caso de uso: reducir stock (venta)."""
        # Obtiene usando puerto
        entity = await self.repository.obtener(item_id)
        if not entity:
            raise ItemNotFoundError(f"Item {item_id} no existe")
        
        # Aplica lógica de dominio
        entity_actualizada = entity.reducir_stock(cantidad)
        
        # Persiste cambios
        await self.repository.guardar(entity_actualizada)
        
        return entity_actualizada
    
    async def buscar_items(self, criterio: str) -> List[HardwareEntity]:
        """Caso de uso: buscar items."""
        return await self.repository.buscar_por_nombre(criterio)

from domain.ports.repository_port import HardwareRepositoryPort
from domain.entities.hardware import HardwareEntity

class SqlAlchemyHardwareRepository(HardwareRepositoryPort):
    """Implementación del repositorio usando SQLAlchemy."""
    
    def __init__(self, session: Session):
        self.session = session
    
    async def guardar(self, entity: HardwareEntity) -> int:
        # Mapea HardwareEntity (dominio) a HardwareItem (ORM)
        db_item = HardwareItem(
            nombre=entity.nombre,
            categoria=entity.categoria,
            stock=entity.stock,
            precio=entity.precio,
        )
        self.session.add(db_item)
        self.session.commit()
        self.session.refresh(db_item)
        return db_item.id
    
    async def obtener(self, id: int) -> Optional[HardwareEntity]:
        db_item = self.session.query(HardwareItem).filter(
            HardwareItem.id == id
        ).first()
        
        if not db_item:
            return None
        
        # Mapea HardwareItem (ORM) a HardwareEntity (dominio)
        return HardwareEntity(
            nombre=db_item.nombre,
            categoria=db_item.categoria.value,
            stock=db_item.stock,
            precio=db_item.precio,
        )

    repository: HardwareRepositoryPort = SqlAlchemyHardwareRepository(db)
    
    # 2. Crear servicio (inyecta puerto)
    service = HardwareService(repository)
    
    # 3. Llamar caso de uso
    entity = await service.crear_item(
        nombre=request.nombre,
        categoria=request.categoria,
        stock=request.stock,
        precio=request.precio,
    )
    
    # 4. Adaptar respuesta a HTTP
    return HardwareResponse(
        nombre=entity.nombre,
        categoria=entity.categoria,
        stock=entity.stock,
        precio=entity.precio,
    )

@router.get("/items/{item_id}")
async def obtener_item(
    item_id: int,
    db: Session = Depends(get_db),
) -> HardwareResponse:
    """Obtener un item."""
    repository = SqlAlchemyHardwareRepository(db)
    service = HardwareService(repository)
    
    entity = await service.obtener_item(item_id)
    if not entity:
        raise HTTPException(status_code=404)
    
    return HardwareResponse.from_entity(entity)
